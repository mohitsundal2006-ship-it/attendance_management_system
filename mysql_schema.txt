-- ==========================================================
-- SMART COLLEGE ATTENDANCE MANAGEMENT SYSTEM (MySQL 8+)
-- ==========================================================

CREATE DATABASE IF NOT EXISTS smart_college_attendance_system;
USE smart_college_attendance_system;

SET FOREIGN_KEY_CHECKS = 0;

DROP TABLE IF EXISTS D_BAR;
DROP TABLE IF EXISTS Attendance_Summary;
DROP TABLE IF EXISTS Attendance;
DROP TABLE IF EXISTS Classes;
DROP TABLE IF EXISTS Student_Enrollment;
DROP TABLE IF EXISTS Academic_Year;
DROP TABLE IF EXISTS Subjects;
DROP TABLE IF EXISTS Students;
DROP TABLE IF EXISTS Faculty;
DROP TABLE IF EXISTS Departments;

SET FOREIGN_KEY_CHECKS = 1;

-- ==========================================================
-- 1. TABLES
-- ==========================================================

CREATE TABLE Departments (
    department_id INT PRIMARY KEY,
    department_name VARCHAR(100) NOT NULL
);

CREATE TABLE Faculty (
    faculty_id INT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    department_id INT NOT NULL,
    CONSTRAINT fk_faculty_department
        FOREIGN KEY (department_id) REFERENCES Departments(department_id)
        ON UPDATE CASCADE ON DELETE RESTRICT
);

CREATE TABLE Students (
    student_id INT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    year INT NOT NULL,
    section VARCHAR(5) NOT NULL,
    department_id INT NOT NULL,
    mentor_id INT NOT NULL,
    CONSTRAINT fk_students_department
        FOREIGN KEY (department_id) REFERENCES Departments(department_id)
        ON UPDATE CASCADE ON DELETE RESTRICT,
    CONSTRAINT fk_students_mentor
        FOREIGN KEY (mentor_id) REFERENCES Faculty(faculty_id)
        ON UPDATE CASCADE ON DELETE RESTRICT
);

CREATE TABLE Subjects (
    subject_id INT PRIMARY KEY,
    subject_name VARCHAR(100) NOT NULL,
    faculty_id INT NOT NULL,
    department_id INT NOT NULL,
    CONSTRAINT fk_subjects_faculty
        FOREIGN KEY (faculty_id) REFERENCES Faculty(faculty_id)
        ON UPDATE CASCADE ON DELETE RESTRICT,
    CONSTRAINT fk_subjects_department
        FOREIGN KEY (department_id) REFERENCES Departments(department_id)
        ON UPDATE CASCADE ON DELETE RESTRICT
);

CREATE TABLE Academic_Year (
    year_id INT PRIMARY KEY,
    year_name VARCHAR(20) NOT NULL
);

CREATE TABLE Student_Enrollment (
    enroll_id INT PRIMARY KEY,
    student_id INT NOT NULL,
    subject_id INT NOT NULL,
    year_id INT NOT NULL,
    CONSTRAINT fk_enrollment_student
        FOREIGN KEY (student_id) REFERENCES Students(student_id)
        ON UPDATE CASCADE ON DELETE RESTRICT,
    CONSTRAINT fk_enrollment_subject
        FOREIGN KEY (subject_id) REFERENCES Subjects(subject_id)
        ON UPDATE CASCADE ON DELETE RESTRICT,
    CONSTRAINT fk_enrollment_year
        FOREIGN KEY (year_id) REFERENCES Academic_Year(year_id)
        ON UPDATE CASCADE ON DELETE RESTRICT
);

CREATE TABLE Classes (
    class_id INT PRIMARY KEY,
    subject_id INT NOT NULL,
    class_date DATE NOT NULL,
    time_slot VARCHAR(20) NOT NULL,
    year_id INT NOT NULL,
    CONSTRAINT fk_classes_subject
        FOREIGN KEY (subject_id) REFERENCES Subjects(subject_id)
        ON UPDATE CASCADE ON DELETE RESTRICT,
    CONSTRAINT fk_classes_year
        FOREIGN KEY (year_id) REFERENCES Academic_Year(year_id)
        ON UPDATE CASCADE ON DELETE RESTRICT
);

CREATE TABLE Attendance (
    attendance_id INT PRIMARY KEY,
    enroll_id INT NOT NULL,
    class_id INT NOT NULL,
    status VARCHAR(10) NOT NULL,
    CONSTRAINT chk_attendance_status CHECK (status IN ('Present', 'Absent')),
    CONSTRAINT uq_attendance_enroll_class UNIQUE (enroll_id, class_id),
    CONSTRAINT fk_attendance_enroll
        FOREIGN KEY (enroll_id) REFERENCES Student_Enrollment(enroll_id)
        ON UPDATE CASCADE ON DELETE RESTRICT,
    CONSTRAINT fk_attendance_class
        FOREIGN KEY (class_id) REFERENCES Classes(class_id)
        ON UPDATE CASCADE ON DELETE RESTRICT
);

CREATE TABLE Attendance_Summary (
    summary_id INT PRIMARY KEY,
    enroll_id INT NOT NULL UNIQUE,
    total_classes INT NOT NULL DEFAULT 0,
    classes_present INT NOT NULL DEFAULT 0,
    percentage DECIMAL(5,2) NOT NULL DEFAULT 0.00,
    CONSTRAINT chk_attendance_percentage CHECK (percentage >= 0 AND percentage <= 100),
    CONSTRAINT fk_summary_enroll
        FOREIGN KEY (enroll_id) REFERENCES Student_Enrollment(enroll_id)
        ON UPDATE CASCADE ON DELETE RESTRICT
);

CREATE TABLE D_BAR (
    dbar_id INT PRIMARY KEY,
    student_id INT NOT NULL,
    percentage DECIMAL(5,2) NOT NULL,
    alert_message VARCHAR(200) NOT NULL,
    alert_date DATE NOT NULL,
    CONSTRAINT chk_dbar_percentage CHECK (percentage >= 0 AND percentage <= 100),
    CONSTRAINT fk_dbar_student
        FOREIGN KEY (student_id) REFERENCES Students(student_id)
        ON UPDATE CASCADE ON DELETE RESTRICT
);
